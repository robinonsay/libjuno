79template<typename T, const size_t N>
80JUNO_RESULT_T<T> Dequeue(QUEUE_ROOT_T<T, N>& tQueue)
81{
82 JUNO_RESULT_T<T> tResult{JUNO_STATUS_SUCCESS, {}};
83 if(tQueue.zLength > 0)
84 {
85 tResult.tSuccess = tQueue.tArrBuff.tArr[tQueue.iStartIndex];
86 tQueue.iStartIndex = (tQueue.iStartIndex + 1) % N;
87 tQueue.zLength -= 1;
88 return tResult;
89 }
90 tResult.tStatus = JUNO_STATUS_ERR;
91 JUNO_FAIL(tResult.tStatus, tQueue._pfcnFailureHandler, tQueue._pvFailurUserData, "Queue is empty");
92 return tResult;
93}

95template<typename T, const size_t N>
96JUNO_STATUS_T Enqueue(QUEUE_ROOT_T<T, N>& tQueue, T tData)
97{
98 if(tQueue.zLength < N)
99 {
100 tQueue.tArrBuff.tArr[(tQueue.iStartIndex + tQueue.zLength) % N] = tData;
101 tQueue.zLength += 1;
102 return JUNO_STATUS_SUCCESS;
103 }
104 JUNO_FAIL(JUNO_STATUS_INVALID_SIZE_ERROR, tQueue._pfcnFailureHandler, tQueue._pvFailurUserData, "Queue is full");
105 return JUNO_STATUS_INVALID_SIZE_ERROR;
106}
