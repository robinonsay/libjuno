167static inline JUNO_STATUS_T JunoDs_Heap_Heapify(JUNO_DS_HEAP_ROOT_T *ptHeap)
168{
169 JUNO_STATUS_T tStatus = JUNO_STATUS_SUCCESS;;
170 tStatus = JunoDs_Heap_Verify(ptHeap);
171 ASSERT_SUCCESS(tStatus, return tStatus);
172 if(ptHeap->zLength <= 0)
173 {
174 tStatus = JUNO_STATUS_ERR;
175 return tStatus;
176 }
177 JUNO_DS_HEAP_INDEX_OPTION_RESULT_T iIndexResult = JunoDs_Heap_ChildGetParent(ptHeap, ptHeap->zLength);
178 ASSERT_SUCCESS(iIndexResult.tStatus, return iIndexResult.tStatus);
179 if(!iIndexResult.tSuccess.bIsSome)
180 {
181 return tStatus;
182 }
183 size_t iIndex = iIndexResult.tSuccess.tSome;
184 for(size_t i = 0; i <= iIndex; ++i)
185 {
186 size_t iCurrentIndex = iIndex - i;
187 iIndexResult = JunoDs_Heap_ChildGetLeft(ptHeap, iCurrentIndex);
188 ASSERT_SUCCESS(iIndexResult.tStatus, continue);
189 iIndexResult = JunoDs_Heap_ChildGetRight(ptHeap, iCurrentIndex);
190 ASSERT_SUCCESS(iIndexResult.tStatus, continue);
191 tStatus = JunoDs_Heap_SiftDown(ptHeap, iCurrentIndex);
192 ASSERT_SUCCESS(tStatus, return tStatus);
193 }
194 return tStatus;
195}

198static inline JUNO_STATUS_T JunoDs_Heap_Delete(JUNO_DS_HEAP_ROOT_T *ptHeap)
199{
200 JUNO_STATUS_T tStatus = JUNO_STATUS_SUCCESS;;
201 tStatus = JunoDs_Heap_Verify(ptHeap);
202 ASSERT_SUCCESS(tStatus, return tStatus);
203 if(ptHeap->zLength <= 0)
204 {
205 tStatus = JUNO_STATUS_ERR;
206 return tStatus;
207 }
208 tStatus = ptHeap->ptApi->Swap(ptHeap, ptHeap->zLength-1, 0);
209 ASSERT_SUCCESS(tStatus, return tStatus);
210 ptHeap->ptApi->Reset(ptHeap, ptHeap->zLength-1);
211 ASSERT_SUCCESS(tStatus, return tStatus);
212 ptHeap->zLength -= 1;
213 return JunoDs_Heap_SiftDown(ptHeap, 0);
214}
