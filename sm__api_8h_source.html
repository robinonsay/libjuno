87static inline JUNO_STATUS_T JunoSm_StateVerify(JUNO_SM_STATE_ROOT_T *ptSmState)
88{
89 JUNO_ASSERT_EXISTS(ptSmState);
90 JUNO_ASSERT_EXISTS(ptSmState->ptApi && ptSmState->ptSm);
91 if(ptSmState->tOptionNextState.bIsSome)
92 {
93 JUNO_ASSERT_EXISTS(ptSmState->tOptionNextState.tSome);
94 }
95 return JUNO_STATUS_SUCCESS;

99static inline JUNO_STATUS_T JunoSm_StateInit(JUNO_SM_ROOT_T *ptSm, JUNO_SM_STATE_ROOT_T *ptStateRoot, JUNO_SM_STATE_ROOT_T *ptNextState, const JUNO_SM_STATE_API_T *ptStateApi, JUNO_FAILURE_HANDLER_T pfcnFailureHandler, JUNO_USER_DATA_T *pvFailureUserData)
100{
101 JUNO_ASSERT_EXISTS(ptStateRoot && ptStateApi && ptSm);
102 ptStateRoot->ptSm = (JUNO_SM_ROOT_T *) ptSm;
103 ptStateRoot->_pfcnFailureHandler = pfcnFailureHandler;
104 ptStateRoot->_pvFailureUserData = pvFailureUserData;
105 ptStateRoot->ptApi = ptStateApi;
106 if(ptNextState)
107 {
108 ptStateRoot->tOptionNextState.bIsSome = true;
109 ptStateRoot->tOptionNextState.tSome = (JUNO_SM_STATE_ROOT_T *) ptNextState;
110 }
111 else
112 {
113 ptStateRoot->tOptionNextState.bIsSome = false;
114 ptStateRoot->tOptionNextState.tSome = NULL;
115 }
116 return JunoSm_StateVerify(ptStateRoot);
117}

143static inline JUNO_SM_RESULT_OPTION_STATE_T JunoSm_TransitionState(JUNO_SM_ROOT_T *ptSmRoot)
144{
145 JUNO_SM_RESULT_OPTION_STATE_T tResult = JUNO_ERR_RESULT(JUNO_STATUS_ERR,
146 JUNO_NONE_OPTION(NULL)
147 );
148 tResult.tStatus = JunoSm_Verify(ptSmRoot);
149 JUNO_ASSERT_SUCCESS(tResult.tStatus, return tResult);
150 tResult.tStatus = JUNO_STATUS_SUCCESS;
151 JUNO_SM_STATE_ROOT_T *ptCurrentStateRoot = (JUNO_SM_STATE_ROOT_T*) ptSmRoot->ptCurrentState;
152 if(ptCurrentStateRoot->tOptionNextState.bIsSome)
153 {
154 ptSmRoot->ptCurrentState = ptCurrentStateRoot->tOptionNextState.tSome;
155 tResult.tOk.bIsSome = true;
156 tResult.tOk.tSome = ptSmRoot->ptCurrentState;
157 }
158 else
159 {
160 tResult.tOk.bIsSome = false;
161 tResult.tOk.tSome = NULL;
162 }
163 return tResult;
164}
